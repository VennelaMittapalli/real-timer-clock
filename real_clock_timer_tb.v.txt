module tb();
	reg RST;
	reg CLK=1'b0;
	wire [6:0] DIS0,DIS1;
	wire [3:0] an0,an1;

	real_timer_top rtt(RST,CLK,DIS0,DIS1,an0,an1);

	always #1 CLK = ~CLK;

	initial begin
      $monitor("%b %b %b %b",DIS0,DIS1,an0,an1);
		RST = 1'b1;
		#1;
		RST = 1'b0;
		#500;
		$finish;
       end
  
  initial begin
    $dumpfile("dump.vcd");
    $dumpvars();
  end

endmodule
